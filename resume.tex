% Custom section command that creates a thick gray underline
% Usage: \resumesection{Section Name}
% Creates a bold section title with a 3pt lighter gray rule underneath
\newcommand{\resumesection}[1]{
    \vspace{1pt}                                    % Add space before section
    \noindent{\fontsize{13.65}{16}\selectfont\textbf{#1}}\\[-5pt]                    % Bold section title with 5% increase (13pt -> 13.65pt)
    \noindent\textcolor{gray!50}{\rule{\textwidth}{3pt}}\\[-6pt]  % Lighter gray underline, now 3pt thick
    \vspace{-10pt}                                   % SIGNIFICANTLY reduce space below the line
}

% Custom experience entry command for consistent formatting
% Parameters: #1 = Company/Institution, #2 = Date Range, #3 = Job Title, #4 = Location
% Usage: \experienceentry{Company}{Date Range}{Job Title}{Location}
\newcommand{\experienceentry}[4]{
    \vspace{0pt}                                    % Add space before entry
    \noindent{\fontsize{9.9}{12}\selectfont\textbf{#1}} \hfill #2\\               % Company name (left, 10% smaller) and dates (right)
    \noindent{\fontsize{10}{12}\selectfont\textit{#3}} \hfill {\fontsize{10}{12}\selectfont\textit{#4}}\\               % Job title (italic, left) and location (right) with -1pt
    \vspace{0pt}                                    % Add space after entry
}

% Custom command for smaller font bullet points to save space
% Usage: \smallbullet{Bullet point text}
% Reduces font size to 9pt for more compact bullet points (additional -1pt)
\newcommand{\smallbullet}[1]{
    \item {\fontsize{9}{11}\selectfont #1}
}

% Custom command for regular bullet points with -1pt font size
% Usage: \regbullet{Bullet point text}
% Reduces font size to 10pt for regular bullet points and decreases top spacing by -2pt
\newcommand{\regbullet}[1]{
    \vspace{-2pt}%
    \item {\fontsize{10}{12}\selectfont #1}
}

% Custom education entry command for consistent formatting
% Parameters: #1 = University Name, #2 = Degree Details (degree, honors, specialization)
% Usage: \educationentry{University Name}{Degree Details}
\newcommand{\educationentry}[2]{
    \vspace{3pt}                                    % Add more space before entry
    \noindent\textbf{#1}\\                          % University name in bold
    \noindent{\fontsize{10}{12}\selectfont #2}\\    % Degree details in regular font
    \vspace{2pt}                                    % Add space after entry
}
